org 1000			; устанавливаю адрес на 1000
val1: int 158 			; 1000
val2: int 13  			; 1004
val3: int 158 			; 1008
val4: int 195495			; 1012
val5: int 777				; 1016
start: 
	CALL _proga		; прямой переход - вызов подпрограммы по адресу 7000 (и сохранение адреса возврата в стек)
	OUTMEMU val5		; вывод числа 777					(ожидаемый результат: 777)
org 7000
_proga: proc
	LOAD val1		; загрузка числа 158 в стек (в стеке: 158 1023)
	ISUB val2		; вычитание 158 - 13 = 145 (в стеке: 145 1023)
	SAVEP val3		; сохранение вершины стека по адресу 1008 и удаление вершины (в стеке: 1023)
	OUTMEMU val3	; вывод числа по адресу 1008 (ожидаемый результат: 145)
	RET			; возврат по адресу в вершине стека и удаление вершины из стека (стек пуст)
endp _proga
end start
