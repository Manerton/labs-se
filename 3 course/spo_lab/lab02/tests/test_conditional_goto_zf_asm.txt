org 1000			; устанавливаю адрес на 1000
val1: int 158 			; 1000
val2: int 13  			; 1004
val3: int 195495		; 1008
val4: int 777			; 1012
start: LOAD val2		; число 13 по адресу 1004 в стек (в стеке: 13)
LOAD val1		; число 158 по адресу 1000 в стек (в стеке: 158 13)
ISUB val2		; разность 158 - 13 = 145 != 0	(в стеке: 145 13)
JZF ifZero		; если результат вычитания = 0, то переход к 1100
JNZF ifNotZero		; если результат вычитания != 0, то переход к 1200 	(ожидаемый результат: 145)

afterIf: 
	OUTMEMU val4		; вывод числа 777 по адресу 1012			    (ожидаемый результат: 777)
	
XCHG 1			; обмен значений в стеке (в стеке: 13 145)
ISUB val2		; разность 13 - 13 = 0
JZF ifZero2			; если результат вычитания = 0, то переход к 1300	(ожидаемый результат: 0)
JNZF ifNotZero2		; если результат вычитания != 0, то переход к 1400	
	
afterIf2:
	OUTMEMU val2		; вывод числа 13									(ожидаемый результат: 13)

org 1100
ifZero: 
	OUTMEMU val1		; вывод числа 158 по адресу 1000 (сюда не попадем)
	JMP afterIf		; прямой безусловный переход по адресу 1031 
	
org 1200
ifNotZero:
	OUTSTU			; вывод числа 145 из вершины стека 
	JMP afterIf		; прямой безусловный переход по адресу 1031
	
org 1300
ifZero2:
	OUTSTU			; вывод числа 0
	JMP afterIf2		; прямой безусловный переход по адресу 1046 
	
org 1400
ifNotZero2:
	OUTMEMU val3		; вывод числа 195495 (сюда не попадем)
	JMP afterIf2		; прямой безусловный переход по адресу 1046 (сюда не попадем)	
	
end start
